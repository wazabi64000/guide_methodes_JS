60 Méthodes JavaScript Courantes

Méthode Description Exemple
sort()Trie un tableau.[3,1,2].sort(); // [1,2,3]
slice()Extrait une portion d’un tableau ou d’une chaîne.[1,2,3,4].slice(1,3); // [2,3]
splice()Modifie un tableau en ajoutant ou supprimant des éléments.[1,2,3].splice(1,1); // [1,3]
trim()Supprime espaces début/fin d’une chaîne." hi ".trim(); // "hi"
push()Ajoute un élément à la fin d’un tableau.[1,2].push(3); // [1,2,3]
pop()Supprime le dernier élément.[1,2,3].pop(); // [1,2]
shift()Supprime le premier élément.[1,2,3].shift(); // [2,3]
unshift()Ajoute au début.[2,3].unshift(1); // [1,2,3]
map()Crée un nouveau tableau avec fonction appliquée.[1,2,3].map(x=>x*2); // [2,4,6]
filter()Filtre selon condition.[1,2,3].filter(x=>x>1); // [2,3]
reduce()Réduit un tableau à une seule valeur.[1,2,3].reduce((a,b)=>a+b,0); // 6
reduceRight()Réduction de droite à gauche.[1,2,3].reduceRight((a,b)=>a-b); // 0
find()Renvoie le premier élément satisfaisant la condition.[1,2,3].find(x=>x>1); // 2
findIndex()Renvoie l’indice du premier élément trouvé.[1,2,3].findIndex(x=>x>1); // 1
includes()Vérifie la présence d’un élément.[1,2,3].includes(2); // true
indexOf()Renvoie l’indice d’un élément.[1,2,3].indexOf(2); // 1
lastIndexOf()Dernier indice d’un élément.[1,2,2].lastIndexOf(2); // 2
join()Concatène les éléments en chaîne.[1,2,3].join('-'); // "1-2-3"
concat()Concatène tableaux ou chaînes.[1,2].concat([3,4]); // [1,2,3,4]
every()Vérifie si tous respectent la condition.[2,4].every(x=>x%2===0); // true
some()Vérifie si au moins un respecte la condition.[1,2,3].some(x=>x>2); // true
reverse()Inverse l’ordre du tableau.[1,2,3].reverse(); // [3,2,1]
fill()Remplit un tableau avec une valeur.[1,2,3].fill(0); // [0,0,0]
copyWithin()Copie une partie dans le même tableau.[1,2,3,4].copyWithin(0,2); // [3,4,3,4]
flat()Aplatie un tableau.[1,[2,3]].flat(); // [1,2,3]
flatMap()Map puis aplatis.[1,2].flatMap(x=>[x,x*2]); // [1,2,2,4]
split()Divise une chaîne en tableau."a-b-c".split('-'); // ["a","b","c"]
replace()Remplace une portion de chaîne."abc".replace("a","x"); // "xbc"
replaceAll()Remplace toutes occurrences."aaa".replaceAll("a","b"); // "bbb"
charAt()Renvoie le caractère à un indice."abc".charAt(1); // "b"
charCodeAt()Code Unicode du caractère."A".charCodeAt(0); // 65
toUpperCase()Met en majuscules."abc".toUpperCase(); // "ABC"
toLowerCase()Met en minuscules."ABC".toLowerCase(); // "abc"
trimStart()Supprime espaces début." hi".trimStart(); // "hi"
trimEnd()Supprime espaces fin."hi ".trimEnd(); // "hi"
padStart()Remplit début chaîne."5".padStart(3,"0"); // "005"
padEnd()Remplit fin chaîne."5".padEnd(3,"0"); // "500"
repeat()Répète la chaîne."ha".repeat(3); // "hahaha"
startsWith()Vérifie début chaîne."abc".startsWith("a"); // true
endsWith()Vérifie fin chaîne."abc".endsWith("c"); // true
Array.isArray()Vérifie si c’est un tableau.Array.isArray([1]); // true
Object.keys()Renvoie clés d’un objet.Object.keys({a:1}); // ["a"]
Object.values()Renvoie valeurs d’un objet.Object.values({a:1}); // [1]
Object.entries()Renvoie paires [clé,valeur].Object.entries({a:1}); // [["a",1]]
hasOwnProperty()Vérifie si clé existe dans l’objet.{a:1}.hasOwnProperty("a"); // true
forEach()Itère sur tableau.[1,2].forEach(x=>console.log(x));
toString()Convertit en chaîne.[1,2].toString(); // "1,2"
valueOf()Renvoie valeur primitive.(123).valueOf(); // 123
Number()Convertit en nombre.Number("123"); // 123
String()Convertit en chaîne.String(123); // "123"
Boolean()Convertit en booléen.Boolean(0); // false
parseInt()Convertit chaîne en entier.parseInt("42"); // 42
parseFloat()Convertit chaîne en flottant.parseFloat("3.14"); // 3.14
isNaN()Vérifie si valeur n’est pas un nombre.isNaN("abc"); // true
isFinite()Vérifie si nombre fini.isFinite(10/2); // true
Math.max()Renvoie le max.Math.max(1,5,3); // 5
Math.min()Renvoie le min.Math.min(1,5,3); // 1
Math.round()Arrondi au plus proche entier.Math.round(4.6); // 5
Math.floor()Arrondi vers le bas.Math.floor(4.6); // 4
Math.ceil()Arrondi vers le haut.Math.ceil(4.1); // 5
Math.random()Nombre aléatoire 0-1.Math.random(); // 0.732...
Math.abs()Valeur absolue.Math.abs(-5); // 5
